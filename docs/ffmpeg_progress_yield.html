104    def set_stderr_callback(self, callback: Callable[[str], None]) -> None:
105        """
106        Set a callback function to be called on stderr output.
107        The callback function must accept a single string argument.
108        Note that this is called on every line of stderr output, so it can be called a lot.
109        Also note that stdout/stderr are joined into one stream, so you might get stdout output in the callback.
110
111        Args:
112            callback (Callable[[str], None]): A callback function that accepts a single string argument.
113        """
114        if not callable(callback) or len(callback.__code__.co_varnames) != 1:
115            raise ValueError(
116                "Callback must be a function that accepts only one argument"
117            )
118
119        self.stderr_callback = callback

121    def run_command_with_progress(
122        self, popen_kwargs=None, duration_override: Union[float, None] = None
123    ) -> Iterator[float]:
124        """
125        Run an ffmpeg command, trying to capture the process output and calculate
126        the duration / progress.
127        Yields the progress in percent.
128
129        Args:
130            popen_kwargs (dict, optional): A dict to specify extra arguments to the popen call, e.g. { creationflags: CREATE_NO_WINDOW }
131            duration_override (float, optional): The duration in seconds. If not specified, it will be calculated from the ffmpeg output.
132
133        Raises:
134            RuntimeError: If the command fails, an exception is raised.
135
136        Yields:
137            Iterator[float]: A generator that yields the progress in percent.
138        """
139        if self.dry_run:
140            return self.cmd
141
142        total_dur: Union[None, int] = None
143        if _uses_error_loglevel(self.cmd):
144            total_dur = _probe_duration(self.cmd)
145
146        if duration_override is not None:
147            total_dur = int(duration_override * 1000)
148
149        cmd_with_progress = (
150            [self.cmd[0]] + ["-progress", "-", "-nostats"] + self.cmd[1:]
151        )
152
153        stderr = []
154        base_popen_kwargs = self.base_popen_kwargs.copy()
155        if popen_kwargs is not None:
156            base_popen_kwargs.update(popen_kwargs)
157
158        self.process = subprocess.Popen(
159            cmd_with_progress,
160            **base_popen_kwargs,
161        )  # type: ignore
162
163        yield 0
164
165        while True:
166            if self.process.stdout is None:
167                continue
168
169            stderr_line = (
170                self.process.stdout.readline().decode("utf-8", errors="replace").strip()
171            )
172
173            if self.stderr_callback:
174                self.stderr_callback(stderr_line)
175
176            if stderr_line == "" and self.process.poll() is not None:
177                break
178
179            stderr.append(stderr_line.strip())
180
181            self.stderr = "\n".join(stderr)
182
183            # assign the total duration if it was found. this can happen multiple times for multiple inputs,
184            # in which case we have to determine the overall duration by taking the min/max (dependent on -shortest being present)
185            if (
186                total_dur_match := self.DUR_REGEX.search(stderr_line)
187            ) and duration_override is None:
188                total_dur_ms = to_ms(**total_dur_match.groupdict())
189                if total_dur is not None:
190                    total_dur = (
191                        min(total_dur, total_dur_ms)
192                        if "-shortest" in self.cmd
193                        else max(total_dur, total_dur_ms)
194                    )
195                else:
196                    total_dur = total_dur_ms
197
198            if (
199                progress_time := FfmpegProgress.TIME_REGEX.search(stderr_line)
200            ) and total_dur is not None:
201                elapsed_time = to_ms(**progress_time.groupdict())
202                yield min(max(round(elapsed_time / total_dur * 100, 2), 0), 100)
203
204        if self.process is None or self.process.returncode != 0:
205            _pretty_stderr = "\n".join(stderr)
206            raise RuntimeError(f"Error running command {self.cmd}: {_pretty_stderr}")
207
208        yield 100
209        self.process = None

211    def quit_gracefully(self) -> None:
212        """
213        Quit the ffmpeg process by sending 'q'
214
215        Raises:
216            RuntimeError: If no process is found.
217        """
218        if self.process is None:
219            raise RuntimeError("No process found. Did you run the command?")
220
221        self.process.communicate(input=b"q")
222        self.process.kill()
223        self.process = None
